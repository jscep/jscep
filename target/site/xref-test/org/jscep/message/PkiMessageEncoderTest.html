123     private static PKCS10CertificationRequest getCsr(X500Principal subject,
124 	    PublicKey pubKey, PrivateKey priKey, char[] password)
125 	    throws GeneralSecurityException, IOException {
126 	DERPrintableString cpSet = new DERPrintableString(new String(password));
127 	SubjectPublicKeyInfo pkInfo = SubjectPublicKeyInfo.getInstance(pubKey
128 		.getEncoded());
129 
130 	JcaContentSignerBuilder signerBuilder = new JcaContentSignerBuilder(
131 		"SHA1withRSA");

133 	try {
134 	    signer = signerBuilder.build(priKey);
135 	} catch (OperatorCreationException e) {
136 	    IOException ioe = new IOException();

140 	}

143 		X500Name.getInstance(subject.getEncoded()), pkInfo);
144 	builder.addAttribute(PKCSObjectIdentifiers.pkcs_9_at_challengePassword,
145 		cpSet);
