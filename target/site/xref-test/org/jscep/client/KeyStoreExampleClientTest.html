43  			// For the sake of simplicity, we use an optimistic verifier. This has
44  			// no
45  			// place in production code.
46  			DefaultCallbackHandler handler = new DefaultCallbackHandler(
47  					new OptimisticCertificateVerifier());
48  			Client client = new Client(getUrl(), handler);
49  			// Get the capabilities of the SCEP server
50  			Capabilities caps = client.getCaCapabilities();
51  	
52  			// We construct a Bouncy Castle digital signature provider early on,
53  			// so it can be reused later.
54  			JcaContentSignerBuilder signerBuilder;
55  			if (caps.contains(Capability.SHA_1)) {
56  				signerBuilder = new JcaContentSignerBuilder("SHA1withRSA");
57  			} else {
58  				signerBuilder = new JcaContentSignerBuilder("MD5withRSA");
59  			}
60  	
61  			// The following variables are used to represent the SCEP client
62  			KeyPair idPair = KeyPairGenerator.getInstance("RSA").genKeyPair();
63  			X500Name issuer = new X500Name("CN=entity");
64  			BigInteger serial = new BigInteger(16, new SecureRandom());
65  			Calendar cal = Calendar.getInstance();
66  			cal.add(Calendar.DATE, -1);
67  			Date notBefore = cal.getTime();
68  			cal.add(Calendar.DATE, 2);
69  			Date notAfter = cal.getTime();
70  			X500Name subject = issuer;
71  			PublicKey publicKey = idPair.getPublic();
72  			JcaX509v3CertificateBuilder certBuilder = new JcaX509v3CertificateBuilder(
73  					issuer, serial, notBefore, notAfter, subject, publicKey);
74  			X509CertificateHolder idHolder = certBuilder.build(signerBuilder
75  					.build(idPair.getPrivate()));
76  			// Convert Bouncy Castle representation of X509Certificate into
77  			// something usable
78  			X509Certificate id = (X509Certificate) CertificateFactory.getInstance(
79  					"X509").generateCertificate(
80  					new ByteArrayInputStream(idHolder.getEncoded()));
81  	
82  			// The following variables are used to represent the entity being
83  			// enrolled
84  			X500Name entityName = new X500Name("CN=entity");
85  			KeyPair entityPair = KeyPairGenerator.getInstance("RSA").genKeyPair();
86  			SubjectPublicKeyInfo publicKeyInfo = SubjectPublicKeyInfo
87  					.getInstance(entityPair.getPublic().getEncoded());
88  			// Generate the certificate signing request
89  			PKCS10CertificationRequestBuilder csrBuilder = new PKCS10CertificationRequestBuilder(
90  					entityName, publicKeyInfo);
91  			// SCEP servers usually require a challenge password
92  			csrBuilder.addAttribute(
93  					PKCSObjectIdentifiers.pkcs_9_at_challengePassword,
94  					new DERPrintableString(new String("password".toCharArray())));
95  			ContentSigner signer = signerBuilder.build(entityPair.getPrivate());
96  			PKCS10CertificationRequest csr = csrBuilder.build(signer);
97  	
98  			// Send the enrollment request
99  			EnrollmentResponse response = client
100 					.enrol(id, idPair.getPrivate(), csr);
101 			if (response.isFailure()) {
102 				// Our request was rejected!
103 				System.out.println("Failed!");
104 			} else if (response.isPending()) {
105 				// The server hasn't enrolled us, but we should try again.
106 				System.out.println("Pending!");
107 	
108 				X500Principal entityPrincipal = new X500Principal(
109 						entityName.getEncoded());
110 				// We should deal with the response to the poll too. Since this a
111 				// short-lived
112 				// test, we conveniently stop processing here. Usually you'd
113 				// schedule the poll
114 				// to run at some point in the future.
115 	
116 				// It isn't a requirement to use the same ID and private key.
117 				response = client.poll(id, idPair.getPrivate(), entityPrincipal,
118 						response.getTransactionId());
119 			} else if (response.isSuccess()) {
120 				// The entity has been enrolled
121 				System.out.println("Success!");
122 	
123 				// Convert the store to a certificate chain
124 				CertStore store = response.getCertStore();
125 				Collection<? extends Certificate> certs = store
126 						.getCertificates(null);
127 				Certificate[] chain = new Certificate[certs.size()];
128 	
129 				int i = 0;
130 				for (Certificate certificate : certs) {
131 					chain[i++] = certificate;
132 				}
133 	
134 				// Store the entity key and certificate in a key store
135 				KeyStore entityStore = KeyStore.getInstance("JKS");
136 				entityStore.load(null, null);
137 				entityStore.setKeyEntry("entity", entityPair.getPrivate(),
138 						"secret".toCharArray(), chain);
139 				entityStore.store(new ByteArrayOutputStream(),
140 						"secret".toCharArray());
141 			}
