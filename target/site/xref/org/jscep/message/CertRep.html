23  package org.jscep.message;
24  
25  import org.bouncycastle.cms.CMSSignedData;
26  import org.jscep.transaction.FailInfo;
27  import org.jscep.transaction.MessageType;
28  import org.jscep.transaction.Nonce;
29  import org.jscep.transaction.PkiStatus;
30  import org.jscep.transaction.TransactionId;
31  
32  /**
33   * This class represents a <tt>CertRep</tt> <tt>pkiMessage</tt>.
34   */
35  public final class CertRep extends PkiMessage<CMSSignedData> {
36      private final Nonce recipientNonce;
37      private final PkiStatus pkiStatus;
38      private final FailInfo failInfo;
39  
40      /**
41       * Creates a new CertRep to indicate  the original request is in a <em>success</em> state.
42       * 
43       * @param transId
44       *            the transaction ID of the original request.
45       * @param senderNonce
46       *            the nonce of the original request.
47       * @param recipientNonce
48       *            the nonce of the response.
49       * @param messageData
50       *            the message data
51       */
52      public CertRep(TransactionId transId, Nonce senderNonce,
53  	    Nonce recipientNonce, CMSSignedData messageData) {
54  	super(transId, MessageType.CERT_REP, senderNonce, messageData);
55  	this.recipientNonce = recipientNonce;
56  	this.pkiStatus = PkiStatus.SUCCESS;
57  	this.failInfo = null;
58      }
59  
60      /**
61       * Creates a new CertRep to indicate the original request is in a <em>failure</em> state.
62       * 
63       * @param transId
64       *            the transaction ID of the original request.
65       * @param senderNonce
66       *            the nonce of the original request.
67       * @param recipientNonce
68       *            the nonce of the response.
69       * @param failInfo
70       *            the fail info enum
71       */
72      public CertRep(TransactionId transId, Nonce senderNonce,
73  	    Nonce recipientNonce, FailInfo failInfo) {
74  	super(transId, MessageType.CERT_REP, senderNonce, null);
75  	this.recipientNonce = recipientNonce;
76  	this.pkiStatus = PkiStatus.FAILURE;
77  	this.failInfo = failInfo;
78      }
79  
80      /**
81       * Creates a new CertRep to indicate the original request is in a <em>pending</em> state.
82       * 
83       * @param transId
84       *            the transaction ID of the original request.
85       * @param senderNonce
86       *            the nonce of the original request.
87       * @param recipientNonce
88       *            the nonce of the response.
89       */
90      public CertRep(TransactionId transId, Nonce senderNonce,
91  	    Nonce recipientNonce) {
92  	super(transId, MessageType.CERT_REP, senderNonce, null);
93  	this.recipientNonce = recipientNonce;
94  	this.pkiStatus = PkiStatus.PENDING;
95  	this.failInfo = null;
96      }
97  
98      public Nonce getRecipientNonce() {
99          return recipientNonce;
100     }
101 
102     public PkiStatus getPkiStatus() {
103         return pkiStatus;
104     }
105 
106     public FailInfo getFailInfo() {
107         if (pkiStatus != PkiStatus.FAILURE) {
108             throw new IllegalStateException();
109         }
110         return failInfo;
111     }
112 
113     @Override
114     public final CMSSignedData getMessageData() {
115         if (pkiStatus != PkiStatus.SUCCESS) {
116             throw new IllegalStateException();
117         }
118         return super.getMessageData();
119     }
120 }
