56  /**
57   * This class is used to encode a <tt>pkiMessage</tt> into a PKCS #7 signedData
58   * object.
59   * 
60   * @see PkiMessageDecoder
61   */
62  public final class PkiMessageEncoder {
63  	private static final String DATA = "1.2.840.113549.1.7.1";
64  	private static final Logger LOGGER = LoggerFactory
65  			.getLogger(PkiMessageEncoder.class);
66  	private final PrivateKey signerKey;
67  	private final X509Certificate signerId;
68  	private final PkcsPkiEnvelopeEncoder enveloper;
69  	private final String signatureAlgorithm;

72  	 * Creates a new <tt>PkiMessageEncoder</tt> instance.
73  	 * 
74  	 * @param signerKey

103 		this.enveloper = enveloper;
104 		this.signatureAlgorithm = signatureAlgorithm;
105 	}

107 	/**
108 	 * Encodes the provided <tt>PkiMessage</tt> into a PKCS #7
109 	 * <tt>signedData</tt>.
110 	 * 
111 	 * @param message
112 	 *            the <tt>PkiMessage</tt> to encode.
113 	 * @return the encoded <tt>signedData</tt>
114 	 * @throws MessageEncodingException
115 	 *             if there is a problem encoding the <tt>PkiMessage</tt>
116 	 */
117 	public CMSSignedData encode(PkiMessage<?> message)
118 			throws MessageEncodingException {
119 		LOGGER.debug("Encoding pkiMessage");
120 		LOGGER.debug("Encoding message: {}", message);
121 
122 		CMSProcessable content = getContent(message);
123 		LOGGER.debug(
124 				"Signing pkiMessage using key belonging to [issuer={}; serial={}]",
125 				signerId.getIssuerDN(), signerId.getSerialNumber());
126 		try {
127 			CMSSignedDataGenerator generator = new CMSSignedDataGenerator();
128 			generator.addSignerInfoGenerator(getSignerInfo(message));
129 			generator.addCertificates(getCertificates());
130 			LOGGER.debug("Signing {} content", content);
131 			CMSSignedData pkiMessage = generator.generate(DATA, content, true,
132 					(Provider) null, true);
133 			LOGGER.debug("Finished encoding pkiMessage");
134 
135 			return pkiMessage;
136 		} catch (CMSException e) {
137 			throw new MessageEncodingException(e);
138 		} catch (Exception e) {
139 			throw new MessageEncodingException(e);
140 		}
141 	}

143 	private CMSProcessable getContent(PkiMessage<?> message)
144 			throws MessageEncodingException {
145 		CMSProcessable signable;
146 
147 		boolean hasMessageData = true;
148 		if (message instanceof CertRep) {
149 			CertRep response = (CertRep) message;
150 			if (response.getPkiStatus() != PkiStatus.SUCCESS) {
151 				hasMessageData = false;
152 			}
153 		}
154 		if (hasMessageData) {
155 			try {
156 				CMSEnvelopedData ed = encodeMessage(message);
157 				signable = new CMSProcessableByteArray(ed.getEncoded());
158 			} catch (IOException e) {
159 				throw new MessageEncodingException(e);
160 			}
161 		} else {
162 			signable = new CMSAbsentContent();
163 		}
164 		return signable;
165 	}
166 
167 	private CMSEnvelopedData encodeMessage(PkiMessage<?> message)
168 			throws MessageEncodingException {
169 		Object messageData = message.getMessageData();
170 		byte[] bytes;
171 		if (messageData instanceof byte[]) {
172 			bytes = (byte[]) messageData;
173 		} else if (messageData instanceof PKCS10CertificationRequest) {
174 			try {
175 				bytes = ((PKCS10CertificationRequest) messageData).getEncoded();
176 			} catch (IOException e) {
177 				throw new MessageEncodingException(e);
178 			}
179 		} else if (messageData instanceof CMSSignedData) {
180 			try {
181 				bytes = ((CMSSignedData) messageData).getEncoded();
182 			} catch (IOException e) {
183 				throw new MessageEncodingException(e);
184 			}
185 		} else {
186 			try {
187 				bytes = ((ASN1Object) messageData).getEncoded();
188 			} catch (IOException e) {
189 				throw new MessageEncodingException(e);
190 			}
191 		}
192 		return enveloper.encode(bytes);
193 	}

195 	private JcaCertStore getCertificates() throws MessageEncodingException {
196 		Collection<X509Certificate> certColl = Collections.singleton(signerId);
197 		JcaCertStore certStore;
198 		try {
199 			certStore = new JcaCertStore(certColl);
200 		} catch (CertificateEncodingException e) {
201 			throw new MessageEncodingException(e);
202 		}
203 		return certStore;
204 	}

206 	private SignerInfoGenerator getSignerInfo(PkiMessage<?> message)
207 			throws MessageEncodingException {
208 		JcaSignerInfoGeneratorBuilder signerInfoBuilder = new JcaSignerInfoGeneratorBuilder(
209 				getDigestCalculator());
210 		signerInfoBuilder
211 				.setSignedAttributeGenerator(getTableGenerator(message));
212 		SignerInfoGenerator signerInfo;
213 		try {
214 			signerInfo = signerInfoBuilder.build(getContentSigner(), signerId);
215 		} catch (Exception e) {
216 			throw new MessageEncodingException(e);
217 		}
218 		return signerInfo;
219 	}
220 
221 	private CMSAttributeTableGenerator getTableGenerator(PkiMessage<?> message) {
222 		AttributeTableFactory attrFactory = new AttributeTableFactory();
223 		AttributeTable signedAttrs = attrFactory.fromPkiMessage(message);
224 		CMSAttributeTableGenerator atGen = new DefaultSignedAttributeTableGenerator(
225 				signedAttrs);
226 		return atGen;
227 	}

229 	private DigestCalculatorProvider getDigestCalculator()
230 			throws MessageEncodingException {
231 		try {
232 			return new JcaDigestCalculatorProviderBuilder().build();
233 		} catch (OperatorCreationException e) {
234 			throw new MessageEncodingException(e);
235 		}
236 	}

238 	private ContentSigner getContentSigner() throws OperatorCreationException {
239 		return new JcaContentSignerBuilder(signatureAlgorithm).build(signerKey);
240 	}
