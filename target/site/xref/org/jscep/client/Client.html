82  /**
83   * The <tt>Client</tt> class is used for interacting with a SCEP server.
84   * <p>
85   * Typical usage might look like so:
86   * 
87   * <pre>
88   * // Create the client
89   * URL server = new URL(&quot;http://jscep.org/scep/pkiclient.exe&quot;);
90   * CertificateVerifier verifier = new ConsoleCertificateVerifier();
91   * Client client = new Client(server, verifier);
92   * 
93   * // Invoke operations on the client.
94   * client.getCaCapabilities();
95   * </pre>
96   * 
97   * Each of the operations of this class is overloaded with a profile argument to
98   * support SCEP servers with multiple (or mandatory) profile names.
99   */
100 public final class Client {
101 	private static final Logger LOGGER = LoggerFactory.getLogger(Client.class);

103 	// A requester MUST have the following information locally configured:
104 	//
105 	// 1. The Certification Authority IP address or fully qualified domain name
106 	// 2. The Certification Authority HTTP CGI script path
107 	//
108 	// We use a URL for this.
109 	private final URL url;
110 	// A requester MUST have the following information locally configured:
111 	//
112 	// 3. The identifying information that is used for authentication of the
113 	// Certification Authority in Section 4.1.1. This information MAY be
114 	// obtained from the user, or presented to the end user for manual
115 	// authorization during the protocol exchange (e.g. the user indicates
116 	// acceptance of a fingerprint via a user-interface element).
117 	//
118 	// We use a callback handler for this.
119 	private final CallbackHandler handler;

121 	/**
122 	 * Constructs a new <tt>Client</tt> instance using the provided
123 	 * <tt>CallbackHandler</tt> for the provided URL.
124 	 * <p>
125 	 * The <tt>CallbackHandler</tt> must be able to handle
126 	 * {@link CertificateVerificationCallback}. Unless the
127 	 * <tt>CallbackHandler</tt> will be used to handle additional
128 	 * <tt>Callback</tt>s, users of this class are recommended to use the
129 	 * {@link #Client(URL, CertificateVerifier)} constructor instead.
130 	 * 
131 	 * @param url
132 	 *            the URL of the SCEP server.
133 	 * @param handler
134 	 *            the callback handler used to check the CA identity.
135 	 */
136 	public Client(URL url, CallbackHandler handler) {
137 		this.url = url;
138 		this.handler = handler;
139 
140 		validateInput();
141 	}

162 	/**
163 	 * Validates all the input to this client.
164 	 * 
165 	 * @throws NullPointerException
166 	 *             if any member variables are null.
167 	 * @throws IllegalArgumentException
168 	 *             if any member variables are invalid.
169 	 */
170 	private void validateInput() throws NullPointerException,
171 			IllegalArgumentException {
172 		// Check for null values first.
173 		if (url == null) {
174 			throw new NullPointerException("URL should not be null");
175 		}
176 		if (!url.getProtocol().matches("^https?$")) {
177 			throw new IllegalArgumentException(
178 					"URL protocol should be HTTP or HTTPS");
179 		}
180 		if (url.getRef() != null) {
181 			throw new IllegalArgumentException(
182 					"URL should contain no reference");
183 		}
184 		if (url.getQuery() != null) {
185 			throw new IllegalArgumentException(
186 					"URL should contain no query string");
187 		}
188 		if (handler == null) {
189 			throw new NullPointerException(
190 					"Callback handler should not be null");
191 		}
192 	}

247 	 * Retrieves the certificates used by the SCEP server.
248 	 * <p>
249 	 * This method queries the server for the certificates it will use in a SCEP
250 	 * message exchange. If the SCEP server represents a single entity, only a
251 	 * single CA certificate will be returned. If the SCEP server supports
252 	 * multiple entities (for example, if it uses a separate entity for signing
253 	 * SCEP messages), additional RA certificates will also be returned.
254 	 * <p>
255 	 * This method provides support for SCEP servers with multiple profiles.
256 	 * 
257 	 * @param profile
258 	 *            the SCEP server profile.
259 	 * @return the certificate store.
260 	 * @throws ClientException
261 	 *             if any client error occurs.
262 	 * @see CertStoreInspector
263 	 */
264 	public CertStore getCaCertificate(final String profile)
265 			throws ClientException {
266 		LOGGER.debug("Retrieving current CA certificate");
267 		// NON-TRANSACTIONAL
268 		// CA and RA public key distribution
269 		final GetCaCertRequest req = new GetCaCertRequest(profile);
270 		final Transport trans = new HttpGetTransport(url);
271 
272 		CertStore store;
273 		try {
274 			store = trans.sendRequest(req, new GetCaCertResponseHandler());
275 		} catch (TransportException e) {
276 			throw new ClientException(e);
277 		}
278 		CertStoreInspector certs = CertStoreInspector.getInstance(store);
279 		verifyCA(certs.getIssuer());
280 		verifyRA(certs.getIssuer(), certs.getRecipient());
281 		verifyRA(certs.getIssuer(), certs.getSigner());
282 
283 		return store;
284 	}

286 	private void verifyRA(X509Certificate ca, X509Certificate ra)
287 			throws ClientException {
288 		LOGGER.debug("Verifying signature of RA certificate");
289 		if (ca.equals(ra)) {
290 			LOGGER.debug("RA and CA are identical");
291 
292 			return;
293 		}
294 		try {
295 			JcaX509CertificateHolder raHolder = new JcaX509CertificateHolder(ra);
296 
297 			ContentVerifierProvider verifierProvider = new JcaContentVerifierProviderBuilder()
298 					.build(ca);
299 
300 			if (!raHolder.isSignatureValid(verifierProvider)) {
301 				LOGGER.debug("Signature verification failed for RA.");
302 				throw new ClientException("RA not issued by CA");
303 			} else {
304 				LOGGER.debug("Signature verification passed for RA.");
305 			}
306 		} catch (CertException e) {
307 			throw new ClientException(e);
308 		} catch (CertificateEncodingException e) {
309 			throw new ClientException(e);
310 		} catch (OperatorCreationException e) {
311 			throw new ClientException(e);
312 		}
313 	}

396 	/**
397 	 * Returns the certificate revocation list a given issuer and serial number.
398 	 * <p>
399 	 * This method requests a CRL for a certificate as identified by the issuer
400 	 * name and the certificate serial number.
401 	 * <p>
402 	 * This method provides support for SCEP servers with multiple profiles.
403 	 * 
404 	 * @param identity
405 	 *            the identity of the client.
406 	 * @param key
407 	 *            the private key to sign the SCEP request.
408 	 * @param issuer
409 	 *            the name of the certificate issuer.
410 	 * @param serial
411 	 *            the serial number of the certificate.
412 	 * @param profile
413 	 *            the SCEP server profile.
414 	 * @return the CRL corresponding to the issuer and serial.
415 	 * @throws ClientException
416 	 *             if any client errors occurs.
417 	 * @throws OperationFailureException
418 	 *             if the request fails.
419 	 */
420 	@SuppressWarnings("unchecked")
421 	public X509CRL getRevocationList(X509Certificate identity, PrivateKey key,
422 			final X500Principal issuer, final BigInteger serial,
423 			final String profile) throws ClientException,
424 			OperationFailureException {
425 		LOGGER.debug("Retriving CRL from CA");
426 		// TRANSACTIONAL
427 		// CRL query
428 		checkDistributionPoints(profile);
429 
430 		X500Name name = new X500Name(issuer.getName());
431 		IssuerAndSerialNumber iasn = new IssuerAndSerialNumber(name, serial);
432 		Transport transport = createTransport(profile);
433 		final Transaction t = new NonEnrollmentTransaction(transport,
434 				getEncoder(identity, key, profile), getDecoder(identity, key,
435 						profile), iasn, MessageType.GET_CRL);
436 		State state;
437 		try {
438 			state = t.send();
439 		} catch (TransactionException e) {
440 			throw new ClientException(e);
441 		}
442 
443 		if (state == State.CERT_ISSUED) {
444 			try {
445 				Collection<X509CRL> crls = (Collection<X509CRL>) t
446 						.getCertStore().getCRLs(null);
447 				if (crls.size() == 0) {
448 					return null;
449 				}
450 				return crls.iterator().next();
451 			} catch (CertStoreException e) {
452 				throw new RuntimeException(e);
453 			}
454 		} else if (state == State.CERT_REQ_PENDING) {
455 			throw new IllegalStateException();
456 		} else {
457 			throw new OperationFailureException(t.getFailInfo());
458 		}
459 	}

461 	private void checkDistributionPoints(final String profile)
462 			throws ClientException {
463 		CertStore store = getCaCertificate(profile);
464 		CertStoreInspector certs = CertStoreInspector.getInstance(store);
465 		final X509Certificate ca = certs.getIssuer();
466 		if (ca.getExtensionValue(X509Extension.cRLDistributionPoints.getId()) != null) {
467 			LOGGER.warn("CA supports distribution points");
468 		}
469 	}

497 	 * Retrieves the certificate corresponding to the provided serial number.
498 	 * <p>
499 	 * This request relates only to the current CA certificate. If the CA
500 	 * certificate has changed since the requested certificate was issued, this
501 	 * operation will fail.
502 	 * <p>
503 	 * This method provides support for SCEP servers with multiple profiles.
504 	 * 
505 	 * @param identity
506 	 *            the identity of the client.
507 	 * @param key
508 	 *            the private key to sign the SCEP request.
509 	 * @param serial
510 	 *            the serial number of the requested certificate.
511 	 * @param profile
512 	 *            the SCEP server profile.
513 	 * @return the certificate store containing the requested certificate.
514 	 * @throws ClientException
515 	 *             if any client error occurs.
516 	 * @throws OperationFailureException
517 	 *             if the SCEP server refuses to service the request.
518 	 */
519 	public CertStore getCertificate(X509Certificate identity, PrivateKey key,
520 			BigInteger serial, String profile)
521 			throws OperationFailureException, ClientException {
522 		LOGGER.debug("Retriving certificate from CA");
523 		// TRANSACTIONAL
524 		// Certificate query
525 		final CertStore store = getCaCertificate(profile);
526 		CertStoreInspector certs = CertStoreInspector.getInstance(store);
527 		final X509Certificate ca = certs.getIssuer();
528 
529 		X500Name name = new X500Name(ca.getIssuerX500Principal().toString());
530 		IssuerAndSerialNumber iasn = new IssuerAndSerialNumber(name, serial);
531 		Transport transport = createTransport(profile);
532 		final Transaction t = new NonEnrollmentTransaction(transport,
533 				getEncoder(identity, key, profile), getDecoder(identity, key,
534 						profile), iasn, MessageType.GET_CERT);
535 
536 		State state;
537 		try {
538 			state = t.send();
539 		} catch (TransactionException e) {
540 			throw new ClientException(e);
541 		}
542 
543 		if (state == State.CERT_ISSUED) {
544 			return t.getCertStore();
545 		} else if (state == State.CERT_REQ_PENDING) {
546 			throw new IllegalStateException();
547 		} else {
548 			throw new OperationFailureException(t.getFailInfo());
549 		}
550 	}

581 	/**
582 	 * Sends a CSR to the SCEP server for enrolling in a PKI.
583 	 * <p>
584 	 * This method enrols the provider <tt>CertificationRequest</tt> into the
585 	 * PKI represented by the SCEP server.
586 	 * 
587 	 * @param identity
588 	 *            the identity of the client.
589 	 * @param key
590 	 *            the private key to sign the SCEP request.
591 	 * @param csr
592 	 *            the CSR to enrol.
593 	 * @param profile
594 	 *            the SCEP server profile.
595 	 * @return the certificate store returned by the server.
596 	 * @throws ClientException
597 	 *             if any client error occurs.
598 	 * @throws OperationFailureException
599 	 *             if the operation fails.
600 	 * @throws PollingTerminatedException
601 	 *             if polling is terminated
602 	 * @throws TransactionException
603 	 *             if there is a problem with the SCEP transaction.
604 	 * @see CertStoreInspector
605 	 */
606 	public EnrollmentResponse enrol(X509Certificate identity, PrivateKey key,
607 			final PKCS10CertificationRequest csr, String profile)
608 			throws ClientException, TransactionException {
609 		LOGGER.debug("Enrolling certificate with CA");
610 
611 		if (isSelfSigned(identity)) {
612 			LOGGER.debug("Certificate is self-signed");
613 			X500Name csrSubject = csr.getSubject();
614 			X500Name idSubject = X500Utils.toX500Name(identity
615 					.getSubjectX500Principal());
616 
617 			if (!csrSubject.equals(idSubject)) {
618 				LOGGER.error("The self-signed certificate MUST use the same subject name as in the PKCS#10 request.");
619 			}
620 		}
621 		// TRANSACTIONAL
622 		// Certificate enrollment
623 		final Transport transport = createTransport(profile);
624 		PkiMessageEncoder encoder = getEncoder(identity, key, profile);
625 		PkiMessageDecoder decoder = getDecoder(identity, key, profile);
626 		final EnrollmentTransaction trans = new EnrollmentTransaction(
627 				transport, encoder, decoder, csr);
628 
629 		try {
630 			MessageDigest digest = getCaCapabilities(profile)
631 					.getStrongestMessageDigest();
632 			byte[] hash = digest.digest(csr.getEncoded());
633 
634 			LOGGER.info("{} PKCS#10 Fingerprint: [{}]", digest.getAlgorithm(),
635 					Hex.encodeHexString(hash));
636 		} catch (IOException e) {
637 			LOGGER.error("Error getting encoded CSR", e);
638 		}
639 
640 		return send(trans);
641 	}

643 	private boolean isSelfSigned(X509Certificate cert) throws ClientException {
644 		try {
645 			JcaX509CertificateHolder holder = new JcaX509CertificateHolder(cert);
646 			ContentVerifierProvider verifierProvider = new JcaContentVerifierProviderBuilder()
647 					.build(holder);
648 
649 			return holder.isSignatureValid(verifierProvider);

682 			throws TransactionException {
683 		State s = trans.send();
684 
685 		if (s == State.CERT_ISSUED) {

687 		} else if (s == State.CERT_REQ_PENDING) {
