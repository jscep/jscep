23  package org.jscep.client;
24  
25  import javax.security.auth.callback.Callback;
26  import java.security.cert.X509Certificate;
27  import java.util.concurrent.atomic.AtomicBoolean;
28  
29  /**
30   * This class is used to obtain verification of a CA certificate.
31   * 
32   * @author David Grant
33   */
34  public final class CertificateVerificationCallback implements Callback {
35      /**
36       * The certificate to verify.
37       */
38      private final X509Certificate caCertificate;
39      /**
40       * The verification status.
41       */
42      private AtomicBoolean verified = new AtomicBoolean(false);
43  
44      /**
45       * Construct a <code>CertificateVerificationCallback</code> with the CA
46       * certificate.
47       * 
48       * @param caCertificate
49       *            the CA certificate to verify.
50       */
51      public CertificateVerificationCallback(final X509Certificate caCertificate) {
52  	this.caCertificate = caCertificate;
53      }
54  
55      /**
56       * Returns the CA certificate to verify.
57       * 
58       * @return the CA certificate to verify.
59       */
60      public X509Certificate getCertificate() {
61  	return caCertificate;
62      }
63  
64      /**
65       * Returns the outcome of the verification process.
66       * 
67       * @return <tt>true</tt> if the certificate was verified, <tt>false</tt>
68       *         otherwise.
69       */
70      public boolean isVerified() {
71  	return verified.get();
72      }
73  
74      /**
75       * Sets the outcome of the callback.
76       * <p/>
77       * {@link #isVerified()} returns <tt>false</tt> by default, so this method
78       * only needs to be called if the CA certificate can be successfully
79       * verified.
80       * 
81       * @param verified
82       *            the outcome.
83       */
84      public void setVerified(final boolean verified) {
85  	this.verified.set(verified);
86      }
87  }
