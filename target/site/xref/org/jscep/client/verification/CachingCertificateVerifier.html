26      }

28      /**
29       * This implementation will forward the given certificate to the delegate
30       * provided in the constructor, and cache the delegate's response.
31       * <p>
32       * On every subsequent invocation with the same certificate, the initial
33       * response from the delegate will be returned.
34       * 
35       * @param cert
36       *            the certificate to verify.
37       * @return the result of calling <tt>verify</tt> on the delegate
38       *         <tt>CertificateVerifier</tt>.
39       */
40      public synchronized boolean verify(final X509Certificate cert) {
41  	if (verificationAnswers.containsKey(cert)) {
42  	    return verificationAnswers.get(cert);
43  	} else {
44  	    boolean answer = delegate.verify(cert);
45  	    verificationAnswers.put(cert, answer);
46  
47  	    return answer;
48  	}
49      }
