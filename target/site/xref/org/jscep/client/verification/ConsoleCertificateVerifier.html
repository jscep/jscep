25  public final class ConsoleCertificateVerifier implements CertificateVerifier {

27      /**
28       * Use the system console to prompt the user with each available hash.
29       * 
30       * @param cert
31       *            the certificate to verify
32       * @return <tt>true</tt> if the certificate is verified, <tt>false</tt>
33       *         otherwise.
34       */
35      public boolean verify(final X509Certificate cert) {
36  	final List<String> algs = new ArrayList<String>(
37  		getAlgorithms(MessageDigest.class.getSimpleName()));
38  	Collections.sort(algs);
39  	int max = 0;
40  	for (final String alg : algs) {
41  	    if (alg.length() > max) {
42  		max = alg.length();
43  	    }
44  	}
45  	for (final String alg : algs) {
46  	    MessageDigest digest;
47  	    try {
48  		digest = MessageDigest.getInstance(alg);
49  	    } catch (NoSuchAlgorithmException e) {
50  		return false;
51  	    }
52  	    byte[] hash;
53  	    try {
54  		hash = digest.digest(cert.getEncoded());
55  	    } catch (CertificateEncodingException e) {
56  		return false;
57  	    }
58  	    System.out.format("%" + max + "s: %s%n", alg,
59  		    Hex.encodeHexString(hash));
60  	}
61  	final Scanner scanner = new Scanner(System.in, Charset.defaultCharset()
62  		.name()).useDelimiter(String.format("%n"));
63  	while (true) {
64  	    System.out.format("%nAccept? (Y/N): [N]%n");
65  	    try {
66  		final String answer = scanner.next();
67  		System.out.println(answer);
68  		if (answer.equals("Y")) {
69  		    return true;
70  		} else if (answer.equals("N")) {
71  		    return false;
72  		}
73  	    } catch (final NoSuchElementException e) {
74  		return false;
75  	    }
76  	}
77      }
