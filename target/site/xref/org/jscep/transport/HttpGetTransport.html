20   * Transport representing the <code>HTTP GET</code> method
21   */
22  @ThreadSafe
23  public final class HttpGetTransport extends Transport {
24      private static final Logger LOGGER = LoggerFactory
25  	    .getLogger(HttpGetTransport.class);
26  
27      /**
28       * Creates a new <tt>HttpGetTransport</tt> for the given <tt>URL</tt>.
29       * 
30       * @param url the <tt>URL</tt> to send <tt>GET</tt> requests to.
31       */
32      public HttpGetTransport(URL url) {
33  	super(url);
34      }
35  
36      /**
37       * {@inheritDoc}
38       */
39      @Override
40      public <T> T sendRequest(Request msg, ScepResponseHandler<T> handler)
41  	    throws TransportException {
42  	URL url = getUrl(msg.getOperation(), msg.getMessage());
43  	if (LOGGER.isDebugEnabled()) {
44  	    LOGGER.debug("Sending {} to {}", msg, url);
45  	}
46  	HttpURLConnection conn;
47  	try {
48  	    conn = (HttpURLConnection) url.openConnection();
49  	} catch (IOException e) {
50  	    throw new TransportException(e);
51  	}
52  
53  	try {
54  	    int responseCode = conn.getResponseCode();
55  	    String responseMessage = conn.getResponseMessage();
56  
57  	    LOGGER.debug("Received '{} {}' when sending {} to {}",
58  		    varargs(responseCode, responseMessage, msg, url));
59  	    if (responseCode != HttpURLConnection.HTTP_OK) {
60  		throw new TransportException(responseCode + " "
61  			+ responseMessage);
62  	    }
63  	} catch (IOException e) {
64  	    throw new TransportException("Error connecting to server", e);
65  	}
66  
67  	byte[] response;
68  	try {
69  	    response = IOUtils.toByteArray(conn.getInputStream());
70  	} catch (IOException e) {
71  	    throw new TransportException("Error reading response stream", e);
72  	}
73  
74  	return handler.getResponse(response, conn.getContentType());
75      }
76  
77      private URL getUrl(Operation op, String message) throws TransportException {
78  	try {
79  	    return new URL(getUrl(op).toExternalForm() + "&message="
80  		    + URLEncoder.encode(message, "UTF-8"));
81  	} catch (MalformedURLException e) {
82  	    throw new TransportException(e);
83  	} catch (UnsupportedEncodingException e) {
84  	    throw new TransportException(e);
85  	}
86      }
