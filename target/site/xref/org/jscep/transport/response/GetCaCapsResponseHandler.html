22  package org.jscep.transport.response;

24  import java.io.BufferedReader;
25  import java.io.ByteArrayInputStream;
26  import java.io.IOException;
27  import java.io.InputStreamReader;
28  import java.nio.charset.Charset;
29  import java.util.EnumSet;
30  import java.util.HashSet;
31  import java.util.Set;
32  
33  import org.apache.commons.io.Charsets;
34  import org.slf4j.Logger;
35  import org.slf4j.LoggerFactory;
36  
37  /**
38   * This class handles responses to <code>GetCACaps</code> requests.
39   * 
40   * @author David Grant
41   */
42  public final class GetCaCapsResponseHandler implements
43  	ScepResponseHandler<Capabilities> {
44      private static final String TEXT_PLAIN = "text/plain";
45      private static final Logger LOGGER = LoggerFactory
46  	    .getLogger(GetCaCapsResponseHandler.class);
47  
48      /**
49       * {@inheritDoc}
50       * 
51       * @throws InvalidContentTypeException
52       */
53      public Capabilities getResponse(byte[] content, String mimeType)
54  	    throws ContentException {
55  	if (mimeType == null || !mimeType.startsWith(TEXT_PLAIN)) {
56  	    LOGGER.warn(
57  		    "Content-Type mismatch: was '{}', expected 'text/plain'",
58  		    mimeType);
59  	}
60  
61  	final EnumSet<Capability> caps = EnumSet.noneOf(Capability.class);
62  
63  	if (LOGGER.isDebugEnabled()) {
64  	    LOGGER.debug("CA capabilities:");
65  	}
66  	BufferedReader reader = new BufferedReader(new InputStreamReader(
67  		new ByteArrayInputStream(content),
68  		Charset.forName(Charsets.US_ASCII.name())));
69  	Set<String> caCaps = new HashSet<String>();
70  	String capability;
71  	try {
72  	    while ((capability = reader.readLine()) != null) {
73  		caCaps.add(capability);
74  	    }
75  	} catch (IOException e) {
76  	    throw new InvalidContentTypeException(e);
77  	} finally {
78  	    try {
79  		reader.close();
80  	    } catch (IOException e) {
81  		LOGGER.error("Error closing reader", e);
82  	    }
83  	}
84  
85  	for (Capability enumValue : Capability.values()) {
86  	    if (caCaps.contains(enumValue.toString())) {
87  		LOGGER.debug("[\u2713] {}", enumValue.getDescription());
88  		caps.add(enumValue);
89  	    } else {
90  		LOGGER.debug("[\u2717] {}", enumValue.getDescription());
91  	    }
92  	}
93  
94  	return new Capabilities(caps.toArray(new Capability[caps.size()]));
95      }
96  }
