18  /**
19   * This class handles responses to <code>GetCACert</code> requests.
20   */
21  public final class GetCaCertResponseHandler implements
22  		ScepResponseHandler<CertStore> {
23  	private static final String RA_CERT = "application/x-x509-ca-ra-cert";
24  	private static final String CA_CERT = "application/x-x509-ca-cert";
25  
26  	/**
27  	 * {@inheritDoc}
28  	 */
29  	public CertStore getResponse(byte[] content, String mimeType)
30  			throws ContentException {
31  		if (mimeType != null && mimeType.startsWith(CA_CERT)) {
32  			// http://tools.ietf.org/html/draft-nourse-scep-20#section-4.1.1.1
33  			CertificateFactory factory;
34  			try {
35  				factory = CertificateFactory.getInstance("X509");
36  			} catch (CertificateException e) {
37  				throw new RuntimeException(e);
38  			}
39  			try {
40  				X509Certificate ca = (X509Certificate) factory
41  						.generateCertificate(new ByteArrayInputStream(content));
42  				Collection<X509Certificate> caSet = Collections.singleton(ca);
43  				CertStoreParameters storeParams = new CollectionCertStoreParameters(
44  						caSet);
45  
46  				return CertStore.getInstance("Collection", storeParams);
47  			} catch (GeneralSecurityException e) {
48  				throw new InvalidContentTypeException(e);
49  			}
50  		} else if (mimeType != null && mimeType.startsWith(RA_CERT)) {
51  			// If an RA is in use, a certificates-only PKCS#7 SignedData
52  			// with a certificate chain consisting of both RA and CA
53  			// certificates is
54  			// returned.
55  			// It should be in the order:
56  			// [0] RA
57  			// [1] CA
58  			if (content.length == 0) {
59  				throw new InvalidContentException(
60  						"Expected a SignedData object, but response was empty");
61  			}
62  			CMSSignedData sd;
63  			try {
64  				sd = new CMSSignedData(content);
65  			} catch (CMSException e) {
66  				throw new InvalidContentException(e);
67  			}
68  			return SignedDataUtils.fromSignedData(sd);
69  		} else {
70  			throw new InvalidContentTypeException(mimeType, CA_CERT, RA_CERT);
71  		}
72  	}
73  }
