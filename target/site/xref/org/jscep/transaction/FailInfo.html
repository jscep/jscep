23  package org.jscep.transaction;
24  
25  /**
26   * This class represents the SCEP <code>failInfo</code> attribute.
27   * 
28   * @author David Grant
29   */
30  public enum FailInfo {
31  	/**
32  	 * Unrecognized or unsupported algorithm identifier
33  	 */
34  	badAlg(0),
35  	/**
36  	 * Integrity check failed
37  	 */
38  	badMessageCheck(1),
39  	/**
40  	 * Transaction not permitted or supported
41  	 */
42  	badRequest(2),
43  	/**
44  	 * The signingTime attribute from the PKCS#7 SignedAttributes was not
45  	 * sufficiently close to the system time
46  	 */
47  	badTime(3),
48  	/**
49  	 * No certificate could be identified matching the provided criteria
50  	 */
51  	badCertId(4);
52  
53  	private final int value;
54  
55  	private FailInfo(int value) {
56  		this.value = value;
57  	}
58  
59  	/**
60  	 * Returns the protocol-specific value for this <tt>failInfo</tt>
61  	 * 
62  	 * @return the protocol-specific value.
63  	 */
64  	public int getValue() {
65  		return value;
66  	}
67  
68  	/**
69  	 * {@inheritDoc}
70  	 */
71  	@Override
72  	public String toString() {
73  		return name();
74  	}
75  
76  	/**
77  	 * Returns the <tt>failInfo</tt> for the given value.
78  	 * <p>
79  	 * If the provided value is not 0-4 inclusive, this method throws an
80  	 * {@link IllegalArgumentException}
81  	 * 
82  	 * @param value
83  	 *            the <tt>failInfo</tt> value.
84  	 * @return the corresponding <tt>failInfo</tt>
85  	 */
86  	public static FailInfo valueOf(int value) {
87  		for (FailInfo failInfo : FailInfo.values()) {
88  			if (failInfo.getValue() == value) {
89  				return failInfo;
90  			}
91  		}
92  		// Fall back to bad request (see issue 39).
93  		return FailInfo.badRequest;
94  	}
95  }
